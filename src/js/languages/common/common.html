import { ObjectIteratee } from '../../../../types';
import { typeAssert } from '../../../error';
import { isObject } from '../../type';


/**
 * Iterates over an object by its own enumerable keys.
 * The iteration can be interrupted by returning `false` by the iteratee function.
 *
 * @param object   - An object to iterate over.
 * @param iteratee - An iteratee function called for each element, taking `value`, `key`, and the current `object`.
 *                   Return `false` explicitly to break the iteration.
 */
export function forOwn<T>( object: T, iteratee: ObjectIteratee<T, void | boolean> ): T {
  isObject( object ) || typeAssert( object );

  const keys = Object.keys( object );

  for ( let i = 0; i < keys.length; i++ ) {
    if ( iteratee( object[ keys[ i ] ], keys[ i ], object ) === false ) {
      break;
    }
  }

  return object;
}
